    <script>

        // 1. YOUR FIREBASE CONFIG 

const firebaseConfig = { 

apiKey: "PASTE_YOUR_API_KEY_HERE", // Found in Project Settings -> General databaseURL: "https://live-polling-msu-prs-default-rtdb.firebaseio.com/", projectId: "live-polling-msu-prs", };


        firebase.initializeApp(firebaseConfig);

        const db = firebase.database();


        let pollOpen = true;


        // 2. LOGIC FOR VOTING

        function castVote(letter) {

            if(!pollOpen) return alert("Poll is closed!");

            db.ref('votes/' + letter).transaction(current => (current || 0) + 1);

            alert("Vote for " + letter + " recorded!");

        }


        // 3. LOGIC FOR PRESENTER

        function togglePoll() {

            pollOpen = !pollOpen;

            const btn = document.getElementById('toggle-poll');

            const results = document.getElementById('results-display');

            

            if(pollOpen) {

                btn.innerText = "Close Polling";

                btn.classList.replace('bg-green-500', 'bg-red-500');

                results.classList.add('opacity-30');

            } else {

                btn.innerText = "Open Polling";

                btn.classList.replace('bg-red-500', 'bg-green-500');

                results.classList.remove('opacity-30');

                updateChart();

            }

            db.ref('settings/status').set(pollOpen);

        }


        // 4. LIVE UPDATES

        function updateChart() {

            db.ref('votes').on('value', (snapshot) => {

                const data = snapshot.val() || {A:0, B:0, C:0, D:0};

                const total = Object.values(data).reduce((a, b) => a + b, 0);

                

                ['A', 'B', 'C', 'D'].forEach(letter => {

                    const count = data[letter] || 0;

                    const percent = total > 0 ? (count / total * 100) : 0;

                    document.getElementById(`count-${letter}`).innerText = count;

                    document.getElementById(`bar-${letter}`).style.width = percent + "%";

                });

            });

        }
